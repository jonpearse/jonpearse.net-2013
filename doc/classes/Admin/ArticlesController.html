def publish
  
  begin
    Article.find(params[:id]).publish!
  rescue e
  end
  
  if params.key?('home')
    respond_to do |format|
      format.html { redirect_to admin_root_path and return }
      format.js
    end
  else
    redirect_to admin_articles_path and return
  end
  
end

def unpublish
  begin
    Article.find(params[:id]).update_attribute(:published, false)
  rescue e
  end
  
  if params.key?('home')
    respond_to do |format|
      format.html { redirect_to admin_root_path and return }
      format.js
    end
  else
    redirect_to admin_articles_path and return
  end
end
