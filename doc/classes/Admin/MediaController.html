def choose_clause( conditions )
  
  if params.key?('type')
    conditions[0] << ' AND media_type = ?'
    conditions << params[:type]
  end
  
  if params.key?('excl')
    conditions[0] << ' AND media_type != ?'
    conditions << params[:excl]
  end
  
end
